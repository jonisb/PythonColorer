    # Replacing %+f, %-f, and % f and specifying a sign:
    s = '{:+f}; {:+f}'.format(3.14, -3.14)  # show it always
    s = '{: f}; {: f}'.format(3.14, -3.14)  # show a space for positive numbers
    s = '{:-f}; {:-f}'.format(3.14, -3.14)  # show only the minus -- same as '{:f}; {:f}'
